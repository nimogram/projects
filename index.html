/** * Nimogram Strategic Sync System (Auto-Sync Mode) * 2026-04-01 09:06 JST */ function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheets()[0]; const lock = LockService.getScriptLock(); try { // 同時書き込みを防ぐロック（10秒） lock.waitLock(10000); // アプリから送られてきたデータを解析 const contents = JSON.parse(e.postData.contents); const incomingData = contents.data || {}; // 1. スプレッドシートから現在のデータを読み込む const currentData = getAllDataFromSheet(sheet); // 2. マージ（統合）ロジック // シートのデータとアプリのデータを統合し、常に「より多くの情報」を持つ状態にします const mergedData = Object.assign({}, currentData, incomingData); // 3. 統合された最新データをシートに保存 saveDataToSheet(sheet, mergedData); // 4. 最新の全データをJSONで返却（Pull機能の核） return ContentService.createTextOutput(JSON.stringify({ status: "success", data: mergedData })).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() })).setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } } function getAllDataFromSheet(sheet) { const lastRow = sheet.getLastRow(); if (lastRow < 2) return {}; const values = sheet.getRange(2, 1, lastRow - 1, 2).getValues(); const data = {}; values.forEach(row => { if (row[0] && row[1]) { try { data[row[0]] = JSON.parse(row[1]); } catch(e) {} } }); return data; } function saveDataToSheet(sheet, mergedData) { sheet.clear(); sheet.getRange(1, 1, 1, 2).setValues([["Date Key", "Slots Data (JSON)"]]); const keys = Object.keys(mergedData).sort(); const rows = keys.map(key => [key, JSON.stringify(mergedData[key])]); if (rows.length > 0) { sheet.getRange(2, 1, rows.length, 2).setValues(rows); } }
